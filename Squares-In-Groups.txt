#include <stdio.h>

int main() {
    int n, i;
    printf("Enter the value of n: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++) {
        // har group me 4 numbers hote hain
        if (i % 4 == 1 || i % 4 == 2) {
            printf("%d ka square = %d\n", i, i * i);
        }
    }

    return 0;
}
